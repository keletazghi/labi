setTimeout(booyah, 2000);
setTimeout(booyah(), 2000);

a, setTimeout(booyah,2000); in this case the function booyah wiil be called after 2 seconds.
b, setTimeout(booyah(),2000); in this case the function will be called without waiting 2 seconds.
